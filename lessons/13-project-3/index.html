generate_drop

import random
from items import HEALTH_POTION, BIG_HEALTH_POTION

DROPS = [HEALTH_POTION, BIG_HEALTH_POTION]

def generate_drop():
    if random.random() < 0.7:   # float в [0, 1)
        return random.choice(DROPS)
    return None

Опыт и уровни

Прогрессия героя

Зачем уровни

  • ощущение роста - победы дают что-то
  • баланс: герой становится сильнее вместе с врагами
  • мотивация продолжать играть
Три новых атрибута: level, xp, xp_threshold

Атрибуты в Character

  • level = 1 - текущий уровень
  • xp = 0 - накопленный опыт
  • xp_threshold = 100 - сколько нужно до следующего уровня
xp_threshold растёт с каждым уровнем - дойти до следующего всё тяжелее

gain_xp и level_up

  • gain_xp(amount) - прибавляем xp, проверяем порог
  • если xp >= xp_threshold - вызываем level_up()
  • level_up: level+1, max_hp+20, damage+5
  • новый порог умножаем на 1.5 - каждый уровень тяжелее
  • при level up - полное лечение

to_dict()

def to_dict(self):
    return {
        "class":         self.__class__.__name__,
        "name":          self.name,
        "level":         self.level,
        "xp":            self.xp,
        "xp_threshold":  self.xp_threshold,
        "gold":          self.gold,
        "hp":            self.hp,
    }
__class__.__name__ даёт строку "Warior" - знаем, какой класс восстанавливать

Сохранить и загрузить

import json

def save_hero(hero, path="save.json"):
    with open(path, "w", encoding="utf-8") as f:
        json.dump(hero.to_dict(), f, ensure_ascii=False)

def load_hero(path="save.json"):
    with open(path, encoding="utf-8") as f:
        data = json.load(f)
    return data  # пока словарь - дальше разберём

from_dict()

@classmethod
def from_dict(cls, data):
    hero = cls(data["name"])
    hero.level         = data["level"]
    hero.xp            = data["xp"]
    hero.xp_threshold  = data["xp_threshold"]
    hero.gold          = data["gold"]
    hero.hp            = data["hp"]
    return hero
@classmethod - вызываем через класс: Warior.from_dict(data)

Выбор класса при загрузке

from characters import Warior, Mage, Archer

CLASSES = {
    "Warior": Warior,
    "Mage":   Mage,
    "Archer": Archer,
}

def load_hero(path="save.json"):
    with open(path, encoding="utf-8") as f:
        data = json.load(f)
    cls = CLASSES[data["class"]]
    return cls.from_dict(data)
